# vuepress 自定义主题搭建自己的博客

# 第一步

在你文档根目录创建一个文件夹blog(自定义) 创建文件README.md 相当于首页

# 全局安装vuepress

yarn global add vuepress # 或者：npm install -g vuepress

# 创建项目目录

mkdir blog
cd blog

# 初始化项目

yarn init -y # 或者 npm init -y

# 运行项目

vuepress dev blog

# 或者在项目目录下局部安装vuepress

yarn add -D vuepress # 或者 npm install -D vuepress

# 配置package.json

{
  "scripts": {
    "dev": "vuepress dev docs",
    "build": "vuepress build docs"
  }
}

# 运行项目

yarn dev # 或者 npm run dev

就初步运行成功啦！

# 文章列表

articles/README.md

//  articles/README.md
---
layout: article_list
---
# 这里是文章列表

// 设置了layout， 默认就会显示`components/article_list.vue`

//  这里是components/article_list.vue
<template>
  <div class="ablock" v-for="index in list.length" :key="list[index-1].key">
    <h1><a :href="list[index-1].path">标题：{{list[index-1].title}}</a></h1>
    <div class="summary">摘要：{{list[index-1].summary}}</div>
    <div class="date">日期：{{list[index-1].frontmatter.date}}</div>
  </div>
</template>

<script>
export default {
  ...
  computed: {
    // 筛选出文章数据，并按frontmatter中的date 排序
    list () {
      return this.$site.pages.filter( v => {
        return v.path.startsWith('/articles/') && v.path.endsWith('html')
      }).sort((x,y) => x.frontmatter.date < y.frontmatter.date)
    }
  }
}
</script>

# 如何根据文章自动生成摘要

上面components/article_list.vue 列表渲染的summary在此处生成

在.vuepress/config.js里配置extendPageData，代码如下（出处）：

const removeMd = require('remove-markdown')
const themeConfig = {
  summary: true,
  summaryLength: 200,
}

module.exports = {
  ...

  extendPageData($page) {
    const strippedContent = $page._strippedContent
    if (!strippedContent) {
      return
    }
    if (themeConfig.summary) {
      $page.summary =
        removeMd(
          strippedContent
            .trim()
            .replace(/^#+\s+(.*)/, '')
            .slice(0, themeConfig.summaryLength)
        ) + ' ...'
      $page.frontmatter.description = $page.summary
    }
    if ($page.frontmatter.summary) {
      $page.frontmatter.description = $page.frontmatter.summary
    }
  }
}

就是把_strippedContent去空格，去掉不必要的字符，截取设定的长度，再通过 remove-markdown去除markdown的格式，然后赋值到$page.summary以及$page.frontmatter.description
